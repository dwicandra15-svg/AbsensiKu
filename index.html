Jalankan di Supabase → SQL Editor:

-- ① TABEL SISWA
CREATE TABLE IF NOT EXISTS siswa (
  id         BIGSERIAL PRIMARY KEY,
  nama       TEXT NOT NULL,
  nisn       TEXT UNIQUE,
  kelas      TEXT,
  rfid_uid   TEXT UNIQUE,
  status     TEXT DEFAULT 'aktif',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ② TABEL ABSENSI
CREATE TABLE IF NOT EXISTS absensi (
  id         BIGSERIAL PRIMARY KEY,
  siswa_id   BIGINT REFERENCES siswa(id) ON DELETE SET NULL,
  rfid_uid   TEXT NOT NULL,
  tipe       TEXT CHECK(tipe IN('masuk','keluar')),
  status     TEXT DEFAULT 'hadir',
  waktu      TIMESTAMPTZ DEFAULT NOW(),
  tanggal    DATE DEFAULT CURRENT_DATE,
  lokasi     TEXT DEFAULT 'Pintu Utama',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ③ TABEL PENGATURAN (branding & konfigurasi — sinkron semua guru)
CREATE TABLE IF NOT EXISTS pengaturan (
  key        TEXT PRIMARY KEY,
  value      TEXT,
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- ④ JIKA TABEL ABSENSI SUDAH ADA — tambah kolom status jika belum ada
ALTER TABLE absensi ADD COLUMN IF NOT EXISTS status TEXT DEFAULT 'hadir';

-- INDEX
CREATE INDEX IF NOT EXISTS ON absensi(tanggal);
CREATE INDEX IF NOT EXISTS ON absensi(siswa_id);
CREATE INDEX IF NOT EXISTS ON siswa(rfid_uid);

-- RLS (akses publik)
ALTER TABLE siswa       ENABLE ROW LEVEL SECURITY;
ALTER TABLE absensi     ENABLE ROW LEVEL SECURITY;
ALTER TABLE pengaturan  ENABLE ROW LEVEL SECURITY;
CREATE POLICY "all" ON siswa       FOR ALL USING(true);
CREATE POLICY "all" ON absensi     FOR ALL USING(true);
CREATE POLICY "all" ON pengaturan  FOR ALL USING(true);
