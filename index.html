SQL Setup Database

Jalankan di Supabase → SQL Editor:

-- TABEL SISWA
CREATE TABLE siswa (
  id         BIGSERIAL PRIMARY KEY,
  nama       TEXT NOT NULL,
  nisn       TEXT UNIQUE,
  kelas      TEXT,
  rfid_uid   TEXT UNIQUE,  -- UID bawaan kartu 13.56MHz
  status     TEXT DEFAULT 'aktif',
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- TABEL ABSENSI
CREATE TABLE absensi (
  id          BIGSERIAL PRIMARY KEY,
  siswa_id    BIGINT REFERENCES siswa(id),
  rfid_uid    TEXT NOT NULL,
  tipe        TEXT CHECK(tipe IN('masuk','keluar')),
  status      TEXT DEFAULT 'hadir',
  -- status: hadir / terlambat / alfa
  waktu       TIMESTAMPTZ DEFAULT NOW(),
  tanggal     DATE DEFAULT CURRENT_DATE,
  lokasi      TEXT DEFAULT 'Pintu Utama',
  created_at  TIMESTAMPTZ DEFAULT NOW()
);

-- INDEX
CREATE INDEX ON absensi(tanggal);
CREATE INDEX ON absensi(siswa_id);
CREATE INDEX ON siswa(rfid_uid);
CREATE INDEX ON siswa(nisn);

-- RLS (akses publik — sesuaikan di prod)
ALTER TABLE siswa   ENABLE ROW LEVEL SECURITY;
ALTER TABLE absensi ENABLE ROW LEVEL SECURITY;
CREATE POLICY "all" ON siswa   FOR ALL USING(true);
CREATE POLICY "all" ON absensi FOR ALL USING(true);
